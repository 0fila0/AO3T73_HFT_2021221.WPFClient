249  public string CheapestShop()
250  {
251  var averagePricesOfShops = from x in this.repo.GetAllProducts()
252  join y in this.repo.GetAllLinks()
253  on x.Termek_ID equals y.Termek_ID
254  join z in this.repo.GetAllShops()
255  on y.Aruhaz_neve equals z.Aruhaz_neve
256  group x by z.Aruhaz_neve into g
257  select new
258  {
259  name = g.Key,
260  averagePrice = g.Average(price => price.Ar),
261  };
262 
263  var thatCheapestShop = averagePricesOfShops.OrderBy(x => x.averagePrice).First();
264 
265  return thatCheapestShop.name;
266  }
