276  public string CheapestShop()

279  join y in this.repo.GetAllLinks()
280  on x.TermekID equals y.TermekID
281  join z in this.repo.GetAllShops()

283  group x by z.AruhazNeve into g
284  select new

287  averagePrice = g.Average(price => price.Ar),
